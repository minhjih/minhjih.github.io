void ShellSort(int a[], int n) {
	int i, j, tmp, t;
	for (t = n / 2; t >= 1; t /= 2) { //gap t 
	// sorting  selection sort, bubble sort     
	// 
		for (i = t; i < n; i++) {// t   
			for (j = i - t; j >= 0; j-=t) { //gap  
			
			    if(a[j+t] > a[j]){//     
			        break;
			    }else{//        .
			        tmp = a[j];
			        a[j] = a[j+t];
			        a[j+t] = tmp;
			    }
			}
		}
	}
}
